196def wrap_complete(
197 governor,
198 controls,
199 attr: Attribution,
200 *,
201 provider: str,
202 model: str,
203 dispatch: DispatchFn,
204 est_input_fn: Callable[[object], int] | None = None,
205 service: str = "",
206 max_call_retries: int = 3,
207):
208 """Wrap a ``complete(provider, model, messages)`` entry point for **pre_call** governance
209 and the **RETRY** actuator.
210
211 After each dispatch the model output is observed (where ``output_runaway`` fires). If a
212 policy sets ``controls.retry`` (degenerate output), the call is re-issued with a tighter
213 output cap and raised frequency/presence penalties — bounded by ``max_call_retries`` and
214 by the policy's own retry budget (it switches to INJECT once exhausted)."""
215 estimate = est_input_fn or _estimate_input_tokens
216 seg = f"run:{attr.run_id}"
217
218 def _observe_llm(response, use_model, messages) -> None:
219 if current_governance() is not None and current_registration() is not None:
220 emit_observation(
221 observation_from_crossing(
222 boundary_id=f"{service or attr.agent}.chat",
223 kind="llm",
224 service=service or attr.agent,
225 input_state={"message_count": len(messages)},
226 result=response,
227 provider=provider,
228 model=use_model,
229 )
230 )
231
232 def governed(p: str, m: str, messages) -> object:
233 controls.begin_call()
234 request = CallRequest(
235 attr=attr, provider=provider, model=m,
236 estimated_input_tokens=estimate(messages),
237 max_output_tokens=controls.call.max_output_tokens,
238 )
239 governor.pre_call(request)
240
241 use_model = controls.call.model_override or m
242 messages = consume_carry(controls, messages)
243 if controls.call.compact: # deep prompt compaction
244 messages = _compact_messages(messages)
245
246 governor.ledger.admit(seg)
247 try:
248 cap = controls.call.max_output_tokens
249 penalties: dict = {}
250 attempt = 0
251 while True:
252 controls.retry = False
253 response = dispatch(p, use_model, messages, max_output_tokens=cap, **penalties)
254 _observe_llm(response, use_model, messages) # may set controls.retry (RETRY)
255 if controls.retry and attempt < max_call_retries:
256 attempt += 1
257 cap = _tighten_cap(cap)
258 penalties = {"frequency_penalty": 1.0, "presence_penalty": 0.6}
259 continue
260 return response
261 finally:
262 governor.ledger.complete(seg)
263
264 return governed
